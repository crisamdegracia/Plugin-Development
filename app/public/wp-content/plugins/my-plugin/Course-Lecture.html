Video 7 - What are action and filter hooks

	
Hooks

		- execute an addtion piece of code or a function
		- modify existing content
		
2 types of hooks

		Action Hooks 
		- Hooks only execute some additional code
		-They dont return anything to the function that called it.
		- without modifying existing information

		Filters
		- Modify existing function
		- Can also erase existing return value
		- We generally pass a variable as a parameter
		
			
			function test(){}
			
			add_action('hook', 'test')
			add_filter('filter', 'test')

Vid 13 - Defining constants

	
		if( ! class_exists( 'MV_Slider') ){
			class MV_slider {
				//Constructor - the first method to be executed;
				function __construct(){
		
					//calling define_constants function
					$this->define_constants();
				}
		
				public function define_constants(){
					//home/www/my_site/wp-content/plugin/your-plugin
					define( 'MV_SLIDER_PATH', plugin_dir_path( __FILE__ ) );
					// http://example.com/wp-content/plugins/mv-slider
					define( 'MV_SLIDER_URL', plugin_dir_url( __FILE__ ) );
					//Plugin version
					define( 'MV_SLIDER_VERSION', '1.0.0' );
				}
			}
		}
